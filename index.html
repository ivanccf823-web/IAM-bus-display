async function updateUI() { if (configs.length === 0) { document.getElementById('display').innerHTML = "請按 ⚙️ 設定"; return; } let html = ''; const now = new Date(); for (const item of configs) { try { let etas = []; if (item.type === 'KMB') { const res = await fetch(`https://data.etabus.gov.hk/v1/transport/kmb/eta/${item.stop_id}/${item.route}/1`); const json = await res.json(); etas = (json.data || []).filter(d => d.dir === item.dir).slice(0, 2).map(d => ({ dest: d.dest_tc, time: d.eta })); } else { const res = await fetch(`https://data.etagmb.gov.hk/eta/route-stop/${item.route_id}/${item.stop_id}`); const json = await res.json(); if (json.data && json.data.eta_res) { etas = json.data.eta_res.slice(0, 2).map(d => ({ dest: '小巴站', time: d.timestamp })); } } // 過濾掉明顯已經過咗去嘅時間 (超過3分鐘前) const validEtas = etas.filter(eta => { const diffSec = (new Date(eta.time) - now) / 1000; return diffSec > -180; // 只要唔係過咗去超過 3 分鐘都仲顯示 }); if (validEtas.length === 0) { html += `
${item.route}服務已結束
明日請早
`; } else { validEtas.forEach(eta => { const diffMs = new Date(eta.time) - now; const diffMin = Math.floor(diffMs / 60000); let timeStr = ''; if (diffMin < 0) { timeStr = '已離開'; } else if (diffMin === 0) { timeStr = '即將抵達'; } else { timeStr = `${diffMin}分`; } html += `
${item.route}往 ${eta.dest}
${timeStr}
`; }); } } catch (e) { addLog(`${item.route} 錯誤: ${e.message}`); } } document.getElementById('display').innerHTML = html; addLog("更新完成"); }
